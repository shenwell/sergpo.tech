{{- /* Слайдер изображений. Строки внутри: путь|||альтернативный текст (тот же текст — подпись под картинкой). Пример: {{< image_slider "Обложки трёх выпусков" >}} /img/a.png|||Описание для alt и подписи /img/b.png|||Другое описание {{< /image_slider >}} */ -}} {{- $label := .Get 0 | default "Галерея изображений" -}} {{- $sid := printf "is-%s" (substr (md5 .Inner) 0 10) -}} {{- $slides := slice -}} {{- range split .Inner "\n" -}} {{- $line := strings.TrimSpace . -}} {{- if $line -}} {{- $parts := split $line "|||" -}} {{- if ne (len $parts) 2 -}} {{- errorf "image_slider: строка должна быть «путь|||alt»: %q" $line -}} {{- end -}} {{- $src := strings.TrimSpace (index $parts 0) -}} {{- $alt := strings.TrimSpace (index $parts 1) -}} {{- $slides = $slides | append (dict "src" $src "alt" $alt) -}} {{- end -}} {{- end -}} {{- if eq (len $slides) 0 -}} {{- errorf "image_slider: нет слайдов (непустые строки путь|||alt)" -}} {{- end -}}
{{- range $i, $s := $slides -}} {{- end -}}
{{- range $i, $_ := $slides -}} {{- end -}}
